async function loadData() {
try {
if (!sb) {
if (typeof window.supabase !== 'undefined') {
sb = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
} else {
// Retry up to 5 times with 1 second delay
let retries = 0;
const retry = async () => {
await new Promise(r => setTimeout(r, 1000));
if (typeof window.supabase !== 'undefined') {
sb = window.supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
} else if (retries++ < 5) {
return retry();
} else {
showToast('Could not connect to cloud — check internet and reload');
return;
}
};
await retry();
if (!sb) return;
}
}

// Load intakes
const {data: intakeRows} = await sb.from('intakes').select('*');
if (intakeRows && intakeRows.length) {
intakes = intakeRows.map(r => r.data);
}

// Load deliveries
const {data: deliveryRows} = await sb.from('deliveries').select('*');
if (deliveryRows && deliveryRows.length) {
deliveries = deliveryRows.filter(r => !r.data._type).map(r => r.data);
dailyRecords = deliveryRows.filter(r => r.data._type === 'daily').map(r => r.data);
}

// Load config (estates, processors, parts, species)
const {data: configRows} = await sb.from('app_config').select('*');
if (configRows) {
for (const row of configRows) {
if (row.key === 'daily_records') dailyRecords = row.data || [];
if (row.key === 'estates' && row.data.length) estates = row.data;
if (row.key === 'processors' && row.data.length) processors = row.data;
if (row.key === 'parts' && row.data.length) parts = row.data;
if (row.key === 'species' && row.data.length) species = row.data;
}
}

// Migration: if Supabase empty, check localStorage
if (intakes.length === 0) {
try {
const raw = localStorage.getItem(KEY);
if (raw) {
const d = JSON.parse(raw);
if (d.intakes && d.intakes.length) {
intakes = d.intakes;
deliveries = d.deliveries || [];
estates = d.estates || [];
processors = d.processors || [];
if (d.parts && d.parts.length) parts = d.parts;
if (d.species && d.species.length) species = d.species;
showToast('Migrating local records to cloud...');
await saveData();
return;
}
}
} catch(e) {}
}

if (intakes.length) showToast(`${intakes.length} record${intakes.length>1?'s':''} · ${estates.length} estate${estates.length!==1?'s':''} · ${processors.length} processor${processors.length!==1?'s':''} loaded`);
} catch(e) {
console.error('Load error', e);
showToast('Could not connect to cloud — check internet connection');
}
}

function seasonCode(dateStr) {
// Sept-Dec: season opens THIS year. Jan-Feb: opened LAST year. Mar-Aug: new season opens THIS year.
const d = dateStr ? new Date(dateStr + 'T12:00:00') : new Date();
const y = d.getFullYear(), m = d.getMonth() + 1;
const openYear = (m >= 9) ? y : (m <= 2) ? y - 1 : y;
const closeYY = String(openYear + 1).slice(-2);
return `${openYear}${closeYY}`;
}

function fmtSeason(c) {
return c && c.length === 6 ? c.slice(0,4) + '/' + c.slice(4) : c;
}

function nextNum(season, list) {
// Match by batchCode prefix — robust against mismatched season fields
const prefix = season + '-';
const rel = (list||[]).filter(i => i.batchCode && i.batchCode.startsWith(prefix));
if (!rel.length) return 1;
const nums = rel.map(i => parseInt(i.batchCode.slice(prefix.length)) || 0);
return Math.max(...nums) + 1;
}

function migrateShortCodes() {
// Fix any records saved with truncated 4-digit season codes e.g. 2526 → 202526
let changed = false;
intakes = intakes.map(r => {
if (r.season && r.season.length === 4) {
const fullSeason = '20' + r.season;
const fixedCode = (r.batchCode||'').startsWith(r.season + '-')
? fullSeason + '-' + r.batchCode.slice(r.season.length + 1)
: r.batchCode;
changed = true;
return {...r, season: fullSeason, batchCode: fixedCode};
}
return r;
});
if (changed) saveData();
}

function seasonFromBatch(batchCode) {
// Extract season from batch code e.g. 202526-3 → 202526
const parts = (batchCode||'').split('-');
const s = parts[0];
return s.length === 6 ? s : null;
}

function describeSeason(code) {
// 202526 → "Sept 2025 – Jan 2026"
if (!code || code.length !== 6) return '';
const openYear = code.slice(0,4);
const closeYY = code.slice(4);
const closeYear = '20' + closeYY;
return `Sept ${openYear} – Jan ${closeYear}`;
}

function tempOk(item) {

return item.storage === 'Chilled' ? t <= 4 : t <= -18;
}

function blankForm() {

function checkDups(f, eid) {
const w = [], o = intakes.filter(r => r.id !== eid);
if (f.batchCode && o.some(r => r.batchCode === f.batchCode)) w.push(`Batch code ${f.batchCode} already exists.`);
if (f.date && f.estateId && f.processorId) {
const c = o.find(r => r.date === f.date && r.estateId === f.estateId && r.processorId === f.processorId);
if (c) w.push(`Same estate & processor already recorded on ${f.date} (${c.batchCode}).`);
}
if (f.ref && f.ref.trim() && o.some(r => r.ref && r.ref.trim() === f.ref.trim() && r.date === f.date))
w.push(`Processor ref "${f.ref}" already used on ${f.date}.`);
return w;
}

function showStatus(s) {

// ── Render helpers ────────────────────────────────────────────────────────────
function el(tag, attrs, ...children) {
const e = document.createElement(tag);
for (const [k,v] of Object.entries(attrs||{})) {
if (k.startsWith('on')) e.addEventListener(k.slice(2).toLowerCase(), v);
else if (k === 'className') e.className = v;
else e.setAttribute(k, v);
}
for (const c of children) {
if (c == null) continue;
e.appendChild(typeof c === 'string' ? document.createTextNode(c) : c);
}
return e;

} else {
item.species = name;
item.custom = '';
}
}
}
const rec = {...form, id: editId || uid()};
if (editId) {
intakes = intakes.map(r => r.id === editId ? rec : r);
} else {
intakes.push(rec);
}

form = null; editId = null;
renderList();
showToast('Record saved');
}

function importData(e) {
const file = e.target.files[0]; if (!file) return;
const reader = new FileReader();
reader.onload = ev => {
try {
const d = JSON.parse(ev.target.result);
if (d.intakes) intakes = d.intakes;
if (d.deliveries) deliveries = d.deliveries;
if (d.estates) estates = d.estates;
if (d.processors) processors = d.processors;
if (d.parts && d.parts.length) parts = d.parts;

}
}
